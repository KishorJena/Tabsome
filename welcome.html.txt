    <script>
        // Open a new tab when the button is clicked
        document.getElementById('openNewTab').addEventListener('click', () => {
            // In a real extension, you might use chrome.tabs.create({});
            // For this standalone page, we'll just show an alert and close the window.
            alert("To see the new tab, simply open a new tab in your browser (Ctrl+T or Cmd+T).");
            window.close();
        });

        // Fun shine effect on mouse move
        const card = document.querySelector('.glass-card');
        const shine = document.getElementById('shine');

        card.addEventListener('mousemove', (e) => {
            const rect = card.getBoundingClientRect();
            const x = e.clientX - rect.left;
            const y = e.clientY - rect.top;
            const rotateX = (y / rect.height - 0.5) * -15; // Invert and reduce angle
            const rotateY = (x / rect.width - 0.5) * 15;
            
            card.style.transform = `perspective(1000px) rotateX(${rotateX}deg) rotateY(${rotateY}deg)`;
            shine.style.transform = `translateX(${x - rect.width/2}px) translateY(${y - rect.height/2}px)`;
        });

        card.addEventListener('mouseleave', () => {
            card.style.transform = 'perspective(1000px) rotateX(0) rotateY(0)';
        });
    </script>
